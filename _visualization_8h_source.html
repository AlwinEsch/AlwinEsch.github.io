725  inline static bool ADDON_OnAction(const AddonInstance_Visualization* addon, VIS_ACTION action, const void *param)
726  {
727  switch (action)
728  {
729  case VIS_ACTION_NEXT_PRESET:
730  return addon->toAddon->addonInstance->NextPreset();
731  case VIS_ACTION_PREV_PRESET:
732  return addon->toAddon->addonInstance->PrevPreset();
733  case VIS_ACTION_LOAD_PRESET:
734  return addon->toAddon->addonInstance->LoadPreset(*static_cast<const int*>(param));
735  case VIS_ACTION_RANDOM_PRESET:
736  return addon->toAddon->addonInstance->RandomPreset();
737  case VIS_ACTION_LOCK_PRESET:
738  addon->toAddon->addonInstance->m_presetLockedByUser = !addon->toAddon->addonInstance->m_presetLockedByUser;
739  return addon->toAddon->addonInstance->LockPreset(addon->toAddon->addonInstance->m_presetLockedByUser);
740  case VIS_ACTION_RATE_PRESET_PLUS:
741  return addon->toAddon->addonInstance->RatePreset(true);
742  case VIS_ACTION_RATE_PRESET_MINUS:
743  return addon->toAddon->addonInstance->RatePreset(false);
744  case VIS_ACTION_UPDATE_ALBUMART:
745  return addon->toAddon->addonInstance->UpdateAlbumart(static_cast<const char*>(param));
746  case VIS_ACTION_UPDATE_TRACK:
747  return addon->toAddon->addonInstance->UpdateTrack(*static_cast<const VisTrack*>(param));
748  case VIS_ACTION_NONE:
749  default:
750  break;
751  }
752  return false;
753  }
