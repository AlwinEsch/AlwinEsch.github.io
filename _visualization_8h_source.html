664  inline static bool ADDON_OnAction(void* addonInstance, VIS_ACTION action, const void *param)
665  {
666  switch (action)
667  {
668  case VIS_ACTION_NEXT_PRESET:
669  return static_cast<CInstanceVisualization*>(addonInstance)->NextPreset();
670  case VIS_ACTION_PREV_PRESET:
671  return static_cast<CInstanceVisualization*>(addonInstance)->PrevPreset();
672  case VIS_ACTION_LOAD_PRESET:
673  return static_cast<CInstanceVisualization*>(addonInstance)->LoadPreset(*(int*)param);
674  case VIS_ACTION_RANDOM_PRESET:
675  return static_cast<CInstanceVisualization*>(addonInstance)->RandomPreset();
676  case VIS_ACTION_LOCK_PRESET:
677  static_cast<CInstanceVisualization*>(addonInstance)->m_presetLockedByUser = !static_cast<CInstanceVisualization*>(addonInstance)->m_presetLockedByUser;
678  return static_cast<CInstanceVisualization*>(addonInstance)->LockPreset(static_cast<CInstanceVisualization*>(addonInstance)->m_presetLockedByUser);
679  case VIS_ACTION_RATE_PRESET_PLUS:
680  return static_cast<CInstanceVisualization*>(addonInstance)->RatePreset(true);
681  case VIS_ACTION_RATE_PRESET_MINUS:
682  return static_cast<CInstanceVisualization*>(addonInstance)->RatePreset(false);
683  case VIS_ACTION_UPDATE_ALBUMART:
684  return static_cast<CInstanceVisualization*>(addonInstance)->UpdateAlbumart(static_cast<const char*>(param));
685  case VIS_ACTION_UPDATE_TRACK:
686  return static_cast<CInstanceVisualization*>(addonInstance)->UpdateTrack(*static_cast<const VisTrack*>(param));
687  case VIS_ACTION_NONE:
688  default:
689  break;
690  }
691  return false;
692  }

701  CInstanceVisualization* addon = static_cast<CInstanceVisualization*>(addonInstance);
702  std::vector<std::string> presets;

704  {
705  for (auto it : presets)
706  addon->m_instance->toKodi.TransferPreset(addon->m_instance->toKodi.kodiInstance, it.c_str());
707  }
708 
709  return presets.size();
710  }
