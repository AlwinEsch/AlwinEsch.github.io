717  inline static bool ADDON_OnAction(const AddonInstance_Visualization* addon, VIS_ACTION action, const void *param)
718  {
719  switch (action)
720  {
721  case VIS_ACTION_NEXT_PRESET:
722  return addon->toAddon.addonInstance->NextPreset();
723  case VIS_ACTION_PREV_PRESET:
724  return addon->toAddon.addonInstance->PrevPreset();
725  case VIS_ACTION_LOAD_PRESET:
726  return addon->toAddon.addonInstance->LoadPreset(*static_cast<const int*>(param));
727  case VIS_ACTION_RANDOM_PRESET:
728  return addon->toAddon.addonInstance->RandomPreset();
729  case VIS_ACTION_LOCK_PRESET:
730  addon->toAddon.addonInstance->m_presetLockedByUser = !addon->toAddon.addonInstance->m_presetLockedByUser;
731  return addon->toAddon.addonInstance->LockPreset(addon->toAddon.addonInstance->m_presetLockedByUser);
732  case VIS_ACTION_RATE_PRESET_PLUS:
733  return addon->toAddon.addonInstance->RatePreset(true);
734  case VIS_ACTION_RATE_PRESET_MINUS:
735  return addon->toAddon.addonInstance->RatePreset(false);
736  case VIS_ACTION_UPDATE_ALBUMART:
737  return addon->toAddon.addonInstance->UpdateAlbumart(static_cast<const char*>(param));
738  case VIS_ACTION_UPDATE_TRACK:
739  return addon->toAddon.addonInstance->UpdateTrack(*static_cast<const VisTrack*>(param));
740  case VIS_ACTION_NONE:
741  default:
742  break;
743  }
744  return false;
745  }
