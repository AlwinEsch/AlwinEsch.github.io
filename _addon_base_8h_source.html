300 /*
301  * Internally used helper class to manage processing of a "C" structure in "CPP"
302  * class.
303  *
304  * At constant, the "C" structure is copied, otherwise the given pointer is
305  * superseded and is changeable.
306  *
307  * -----------------------------------------------------------------------------
308  *
309  * Example:
310  *
311  * ~~~~~~~~~~~~~{.cpp}
312  * extern "C" typedef struct C_SAMPLE_DATA
313  * {
314  * unsigned int iUniqueId;
315  * } C_SAMPLE_DATA;
316  *
317  * class CPPSampleData : public CStructHdl<CPPSampleData, C_SAMPLE_DATA>
318  * {
319  * public:
320  * CPPSampleData() = default;
321  * CPPSampleData(const CPPSampleData& sample) : CStructHdl(sample) { }
322  * CPPSampleData(const C_SAMPLE_DATA* sample) : CStructHdl(sample) { }
323  * CPPSampleData(C_SAMPLE_DATA* sample) : CStructHdl(sample) { }
324  *
325  * void SetUniqueId(unsigned int uniqueId) { m_cStructure->iUniqueId = uniqueId; }
326  * unsigned int GetUniqueId() const { return m_cStructure->iUniqueId; }
327  * };
328  *
329  * ~~~~~~~~~~~~~
330  *
331  * It also works with the following example:
332  *
333  * ~~~~~~~~~~~~~{.cpp}
334  * CPPSampleData test;
335  * // Some work
336  * C_SAMPLE_DATA* data = test;
337  * // Give "data" to Kodi
338  * ~~~~~~~~~~~~~
339  */
340 template<class CPP_CLASS, typename C_STRUCT>
341 class CStructHdl
342 {
343 public:
344  CStructHdl()
345  : m_cStructure(new C_STRUCT())
346  , m_owner(true)
347  {
348  }
349 
350  CStructHdl(const CPP_CLASS& cppClass)
351  : m_cStructure(new C_STRUCT(*cppClass.m_cStructure))
352  , m_owner(true)
353  {
354  }
355 
356  CStructHdl(const C_STRUCT* cStructure)
357  : m_cStructure(new C_STRUCT({*cStructure}))
358  , m_owner(true)
359  {
360  }
361 
362  CStructHdl(C_STRUCT* cStructure)
363  : m_cStructure(cStructure)
364  {
365  assert(cStructure);
366  }
367 
368  const CStructHdl& operator=(const CStructHdl& right)
369  {
370  assert(&right.m_cStructure);
371  if (m_cStructure && !m_owner)
372  {
373  memcpy(m_cStructure, right.m_cStructure, sizeof(C_STRUCT));
374  }
375  else
376  {
377  if (m_owner)
378  delete m_cStructure;
379  m_owner = true;
380  m_cStructure = new C_STRUCT(*right.m_cStructure);
381  }
382  return *this;
383  }
384 
385  const CStructHdl& operator=(const C_STRUCT& right)
386  {
387  assert(&right);
388  if (m_cStructure && !m_owner)
389  {
390  memcpy(m_cStructure, &right, sizeof(C_STRUCT));
391  }
392  else
393  {
394  if (m_owner)
395  delete m_cStructure;
396  m_owner = true;
397  m_cStructure = new C_STRUCT(*right);
398  }
399  return *this;
400  }
401 
402  virtual ~CStructHdl()
403  {
404  if (m_owner)
405  delete m_cStructure;
406  }
407 
408  operator C_STRUCT*() { return m_cStructure; }
409 
410 protected:
411  C_STRUCT* m_cStructure = nullptr;
412 
413 private:
414  bool m_owner = false;
415 };

563  static inline ADDON_STATUS ADDONBASE_CreateInstanceEx(int instanceType, const char* instanceID, KODI_HANDLE instance, KODI_HANDLE* addonInstance, KODI_HANDLE parent, const char* version)
564  {
565  ADDON_STATUS status = ADDON_STATUS_NOT_IMPLEMENTED;
566  if (parent != nullptr)
567  status = static_cast<IAddonInstance*>(parent)->CreateInstanceEx(instanceType, instanceID, instance, *addonInstance, version);
568  if (status == ADDON_STATUS_NOT_IMPLEMENTED)
569  status = CAddonBase::m_interface->addonBase->CreateInstanceEx(instanceType, instanceID, instance, *addonInstance, version);
570  if (*addonInstance == nullptr)
571  throw std::logic_error("kodi::addon::CAddonBase CreateInstanceEx returns a empty instance pointer!");
572 
573  if (static_cast<::kodi::addon::IAddonInstance*>(*addonInstance)->m_type != instanceType)
574  throw std::logic_error("kodi::addon::CAddonBase CreateInstanceEx with difference on given and returned instance type!");
575 
576  // Store the used ID inside instance, to have on destroy calls by addon to identify
577  static_cast<::kodi::addon::IAddonInstance*>(*addonInstance)->m_id = instanceID;
578 
579  return status;
580  }
581 
582  static inline void ADDONBASE_DestroyInstance(int instanceType, KODI_HANDLE instance)
583  {
584  if (CAddonBase::m_interface->globalSingleInstance == nullptr && instance != CAddonBase::m_interface->addonBase)
585  {
586  if (static_cast<::kodi::addon::IAddonInstance*>(instance)->m_type == instanceType)
587  {
588  CAddonBase::m_interface->addonBase->DestroyInstance(
589  instanceType, static_cast<::kodi::addon::IAddonInstance*>(instance)->m_id, instance);
590  delete static_cast<::kodi::addon::IAddonInstance*>(instance);
591  }
592  else
593  throw std::logic_error("kodi::addon::CAddonBase DestroyInstance called with difference on given and present instance type!");
594  }
595  }
