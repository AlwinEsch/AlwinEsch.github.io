154  CAddonSetting(SETTING_TYPE type, std::string id, std::string label)
155  : Type(type),

157  Label(label),
158  Current(0)
159  {

162  CAddonSetting(const CAddonSetting &rhs) // copy constructor
163  {
164  Id = rhs.Id;
165  Label = rhs.Label;
166  Current = rhs.Current;
167  Type = rhs.Type;
168  for (unsigned int i = 0; i < rhs.Entries.size(); ++i)
169  Entries.push_back(rhs.Entries[i]);
170  }
171 
172  void AddEntry(std::string label)
173  {
174  if (Label.empty() || Type != SPIN)
175  return;
176  Entries.push_back(Label);
177  }
178 
179  // data members
180  SETTING_TYPE Type;
181  std::string Id;
182  std::string Label;
183  int Current;

190  //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=
191  // Function used on Kodi itself to transfer back from add-on given data with
192  // "ADDON_StructSetting***" to "std::vector<CAddonSetting>"
193  //
194  // Note: Not needed on add-on itself, only here to have all related parts on
195  // same place!
196  //
197  static inline void StructToVec(unsigned int elements, ADDON_StructSetting*** sSet, std::vector<CAddonSetting> *vecSet)
198  {
199  if (elements == 0)
200  return;
201 
202  vecSet->clear();
203  for(unsigned int i = 0; i < elements; i++)
204  {
205  CAddonSetting vSet((CAddonSetting::SETTING_TYPE)(*sSet)[i]->type, (*sSet)[i]->id, (*sSet)[i]->label);
206  if((*sSet)[i]->type == CAddonSetting::SPIN)
207  {
208  for(unsigned int j=0;j<(*sSet)[i]->entry_elements;j++)
209  {
210  vSet.AddEntry((*sSet)[i]->entry[j]);
211  }
212  }
213  vSet.Current = (*sSet)[i]->current;
214  vecSet->push_back(vSet);
215  }
216  }

270  static inline unsigned int ADDONBASE_GetSettings(ADDON_StructSetting ***sSet)
271  {
272  std::vector<CAddonSetting> settings;
273  if (CAddonBase::m_createdAddon->GetSettings(settings))
274  {
275  *sSet = nullptr;
276  if (settings.empty())
277  return 0;
278 
279  *sSet = (ADDON_StructSetting**)malloc(settings.size()*sizeof(ADDON_StructSetting*));
280  for (unsigned int i = 0;i < settings.size(); ++i)
281  {
282  (*sSet)[i] = nullptr;
283  (*sSet)[i] = (ADDON_StructSetting*)malloc(sizeof(ADDON_StructSetting));
284  (*sSet)[i]->id = nullptr;
285  (*sSet)[i]->label = nullptr;
286 
287  if (!settings[i].Id.empty() && !settings[i].Label.empty())
288  {
289  (*sSet)[i]->id = strdup(settings[i].Id.c_str());
290  (*sSet)[i]->label = strdup(settings[i].Label.c_str());
291  (*sSet)[i]->type = settings[i].Type;
292  (*sSet)[i]->current = settings[i].Current;
293  (*sSet)[i]->entry_elements = 0;
294  (*sSet)[i]->entry = nullptr;
295  if (settings[i].Type == CAddonSetting::SPIN && !settings[i].Entries.empty())
296  {
297  (*sSet)[i]->entry = (char**)malloc(settings[i].Entries.size()*sizeof(char**));
298  for (unsigned int j = 0; j < settings[i].Entries.size(); ++j)
299  {
300  if (!settings[i].Entries[j].empty())
301  {
302  (*sSet)[i]->entry[j] = strdup(settings[i].Entries[j].c_str());
303  (*sSet)[i]->entry_elements++;
304  }
305  }
306  }
307  }
308  }
309 
310  return settings.size();
311  }
312  return 0;
313  }

323  static inline void ADDONBASE_FreeSettings(unsigned int elements, ADDON_StructSetting*** set)
324  {
325  if (elements == 0)
326  return;
327 
328  for (unsigned int i = 0; i < elements; ++i)
329  {
330  if ((*set)[i]->type == CAddonSetting::SPIN)
331  {
332  for (unsigned int j = 0; j < (*set)[i]->entry_elements; ++j)
333  {
334  free((*set)[i]->entry[j]);
335  }
336  free((*set)[i]->entry);
337  }
338  free((*set)[i]->id);
339  free((*set)[i]->label);
340  free((*set)[i]);
341  }
342  free(*set);
343  }
