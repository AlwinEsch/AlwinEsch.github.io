205 /*
206  * Internally used helper class to manage processing of a "C" structure in "CPP"
207  * class.
208  *
209  * At constant, the "C" structure is copied, otherwise the given pointer is
210  * superseded and is changeable.
211  *
212  * -----------------------------------------------------------------------------
213  *
214  * Example:
215  *
216  * ~~~~~~~~~~~~~{.cpp}
217  * extern "C" typedef struct C_SAMPLE_DATA
218  * {
219  * unsigned int iUniqueId;
220  * } C_SAMPLE_DATA;
221  *
222  * class CPPSampleData : public CStructHdl<CPPSampleData, C_SAMPLE_DATA>
223  * {
224  * public:
225  * CPPSampleData() = default;
226  * CPPSampleData(const CPPSampleData& sample) : CStructHdl(sample) { }
227  * CPPSampleData(const C_SAMPLE_DATA* sample) : CStructHdl(sample) { }
228  * CPPSampleData(C_SAMPLE_DATA* sample) : CStructHdl(sample) { }
229  *
230  * void SetUniqueId(unsigned int uniqueId) { m_cStructure->iUniqueId = uniqueId; }
231  * unsigned int GetUniqueId() const { return m_cStructure->iUniqueId; }
232  * };
233  *
234  * ~~~~~~~~~~~~~
235  *
236  * It also works with the following example:
237  *
238  * ~~~~~~~~~~~~~{.cpp}
239  * CPPSampleData test;
240  * // Some work
241  * C_SAMPLE_DATA* data = test;
242  * // Give "data" to Kodi
243  * ~~~~~~~~~~~~~
244  */
245 template<class CPP_CLASS, typename C_STRUCT>
246 class CStructHdl
247 {
248 public:
249  CStructHdl() : m_cStructure(new C_STRUCT()), m_owner(true) {}
250 
251  CStructHdl(const CPP_CLASS& cppClass)
252  : m_cStructure(new C_STRUCT(*cppClass.m_cStructure)), m_owner(true)
253  {
254  }
255 
256  CStructHdl(const C_STRUCT* cStructure) : m_cStructure(new C_STRUCT(*cStructure)), m_owner(true) {}
257 
258  CStructHdl(C_STRUCT* cStructure) : m_cStructure(cStructure) { assert(cStructure); }
259 
260  const CStructHdl& operator=(const CStructHdl& right)
261  {
262  assert(&right.m_cStructure);
263  if (m_cStructure && !m_owner)
264  {
265  memcpy(m_cStructure, right.m_cStructure, sizeof(C_STRUCT));
266  }
267  else
268  {
269  if (m_owner)
270  delete m_cStructure;
271  m_owner = true;
272  m_cStructure = new C_STRUCT(*right.m_cStructure);
273  }
274  return *this;
275  }
276 
277  const CStructHdl& operator=(const C_STRUCT& right)
278  {
279  assert(&right);
280  if (m_cStructure && !m_owner)
281  {
282  memcpy(m_cStructure, &right, sizeof(C_STRUCT));
283  }
284  else
285  {
286  if (m_owner)
287  delete m_cStructure;
288  m_owner = true;
289  m_cStructure = new C_STRUCT(*right);
290  }
291  return *this;
292  }
293 
294  virtual ~CStructHdl()
295  {
296  if (m_owner)
297  delete m_cStructure;
298  }
299 
300  operator C_STRUCT*() { return m_cStructure; }
301  operator const C_STRUCT*() const { return m_cStructure; }
302 
303  const C_STRUCT* GetCStructure() const { return m_cStructure; }
304 
305 protected:
306  C_STRUCT* m_cStructure = nullptr;
307 
308 private:
309  bool m_owner = false;
310 };

503  static inline ADDON_STATUS ADDONBASE_CreateInstance(int instanceType,
504  const char* instanceID,
505  KODI_HANDLE instance,
506  const char* version,
507  KODI_HANDLE* addonInstance,
508  KODI_HANDLE parent)
509  {
510  CAddonBase* base = static_cast<CAddonBase*>(m_interface->addonBase);
511 
512  ADDON_STATUS status = ADDON_STATUS_NOT_IMPLEMENTED;
513 
514  /* Check about single instance usage:
515  * 1. The kodi side instance pointer must be equal to first one
516  * 2. The addon side instance pointer must be set
517  * 3. And the requested type must be equal with used add-on class
518  */
519  if (m_interface->firstKodiInstance == instance && m_interface->globalSingleInstance &&
520  static_cast<IAddonInstance*>(m_interface->globalSingleInstance)->m_type == instanceType)
521  {
522  /* The handling here is intended for the case of the add-on only one
523  * instance and this is integrated in the add-on base class.
524  */
525  *addonInstance = m_interface->globalSingleInstance;
526  status = ADDON_STATUS_OK;
527  }
528  else
529  {
530  /* Here it should use the CreateInstance instance function to allow
531  * creation of several on one addon.
532  */
533 
534  /* Check first a parent is defined about (e.g. Codec within inputstream) */
535  if (parent != nullptr)
536  status = static_cast<IAddonInstance*>(parent)->CreateInstance(
537  instanceType, instanceID, instance, version, *addonInstance);
538 
539  /* if no parent call the main instance creation function to get it */
540  if (status == ADDON_STATUS_NOT_IMPLEMENTED)
541  {
542  status = base->CreateInstance(instanceType, instanceID, instance, version, *addonInstance);
543  }
544  }
545 
546  if (*addonInstance == nullptr)
547  {
548  if (status == ADDON_STATUS_OK)
549  {
550  m_interface->toKodi->addon_log_msg(m_interface->toKodi->kodiBase, ADDON_LOG_FATAL,
551  "kodi::addon::CAddonBase CreateInstance returned an "
552  "empty instance pointer, but reported OK!");
553  return ADDON_STATUS_PERMANENT_FAILURE;
554  }
555  else
556  {
557  return status;
558  }
559  }
560 
561  if (static_cast<IAddonInstance*>(*addonInstance)->m_type != instanceType)
562  {
563  m_interface->toKodi->addon_log_msg(
564  m_interface->toKodi->kodiBase, ADDON_LOG_FATAL,
565  "kodi::addon::CAddonBase CreateInstance difference between given and returned");
566  delete static_cast<IAddonInstance*>(*addonInstance);
567  *addonInstance = nullptr;
568  return ADDON_STATUS_PERMANENT_FAILURE;
569  }
570 
571  // Store the used ID inside instance, to have on destroy calls by addon to identify
572  static_cast<IAddonInstance*>(*addonInstance)->m_id = instanceID;
573 
574  return status;
575  }
576 
577  static inline void ADDONBASE_DestroyInstance(int instanceType, KODI_HANDLE instance)
578  {
579  CAddonBase* base = static_cast<CAddonBase*>(m_interface->addonBase);
580 
581  if (m_interface->globalSingleInstance == nullptr && instance != base)
582  {
583  base->DestroyInstance(instanceType, static_cast<IAddonInstance*>(instance)->m_id, instance);
584  delete static_cast<IAddonInstance*>(instance);
585  }
586  }
