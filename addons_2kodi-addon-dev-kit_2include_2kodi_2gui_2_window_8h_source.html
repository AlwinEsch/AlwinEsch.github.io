860  static void CBGetContextButtons(GUIHANDLE cbhdl, int itemNumber, gui_context_menu_pair* buttons, unsigned int* size)
861  {
862  std::vector< std::pair<unsigned int, std::string> > buttonList;
863  static_cast<CWindow*>(cbhdl)->GetContextButtons(itemNumber, buttonList);
864  if (!buttonList.empty())
865  {
866  unsigned int presentSize = buttonList.size();
867  if (presentSize > *size)
868  kodi::Log(ADDON_LOG_WARNING, "GetContextButtons: More as allowed '%i' entries present!", *size);
869  else
870  *size = presentSize;
871  for (unsigned int i = 0; i < *size; ++i)
872  {
873  buttons[i].id = buttonList[i].first;
874  strncpy(buttons[i].name, buttonList[i].second.c_str(), ADDON_MAX_CONTEXT_ENTRY_NAME_LENGTH);
875  }
876  }
877  }
878 
879  static bool CBOnContextButton(GUIHANDLE cbhdl, int itemNumber, unsigned int button)

881  return static_cast<CWindow*>(cbhdl)->OnContextButton(itemNumber, button);
