853  static void CBGetContextButtons(GUIHANDLE cbhdl, int itemNumber, gui_context_menu_pair* buttons, unsigned int* size)
854  {
855  std::vector< std::pair<unsigned int, std::string> > buttonList;
856  static_cast<CWindow*>(cbhdl)->GetContextButtons(itemNumber, buttonList);
857  if (!buttonList.empty())
858  {
859  unsigned int presentSize = static_cast<unsigned int>(buttonList.size());
860  if (presentSize > *size)
861  kodi::Log(ADDON_LOG_WARNING, "GetContextButtons: More as allowed '%i' entries present!", *size);
862  else
863  *size = presentSize;
864  for (unsigned int i = 0; i < *size; ++i)
865  {
866  buttons[i].id = buttonList[i].first;
867  strncpy(buttons[i].name, buttonList[i].second.c_str(), ADDON_MAX_CONTEXT_ENTRY_NAME_LENGTH);
868  }
869  }
870  }
871
