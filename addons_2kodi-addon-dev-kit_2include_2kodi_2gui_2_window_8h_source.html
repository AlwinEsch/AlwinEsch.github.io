864  static void CBGetContextButtons(GUIHANDLE cbhdl, int itemNumber, gui_context_menu_pair* buttons, unsigned int* size)
865  {
866  std::vector< std::pair<unsigned int, std::string> > buttonList;
867  static_cast<CWindow*>(cbhdl)->GetContextButtons(itemNumber, buttonList);
868  if (!buttonList.empty())
869  {
870  unsigned int presentSize = buttonList.size();
871  if (presentSize > *size)
872  kodi::Log(ADDON_LOG_WARNING, "GetContextButtons: More as allowed '%i' entries present!", *size);
873  else
874  *size = presentSize;
875  for (unsigned int i = 0; i < *size; ++i)
876  {
877  buttons[i].id = buttonList[i].first;
878  strncpy(buttons[i].name, buttonList[i].second.c_str(), ADDON_MAX_CONTEXT_ENTRY_NAME_LENGTH);
879  }
880  }
881  }
882
