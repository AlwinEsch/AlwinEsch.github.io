142  bool IsRunning() const
143  {
144  if (m_thread != nullptr)
145  {
146  // it's possible that the thread exited on it's own without a call to StopThread. If so then
147  // the promise should be fulfilled.
148  std::future_status stat = m_future.wait_for(std::chrono::milliseconds(0));
149  // a status of 'ready' means the future contains the value so the thread has exited
150  // since the thread can't exit without setting the future.
151  if (stat == std::future_status::ready) // this is an indication the thread has exited.
152  return false;
153  return true; // otherwise the thread is still active.
154  }
155  else
156  return false;
157  }

169  void CreateThread(bool autoDelete = false)
170  {
171  if (m_thread != nullptr)
172  {
173  // if the thread exited on it's own, without a call to StopThread, then we can get here
174  // incorrectly. We should be able to determine this by checking the promise.
175  std::future_status stat = m_future.wait_for(std::chrono::milliseconds(0));
176  // a status of 'ready' means the future contains the value so the thread has exited
177  // since the thread can't exit without setting the future.
178  if (stat == std::future_status::ready) // this is an indication the thread has exited.
179  StopThread(true); // so let's just clean up
180  else
181  { // otherwise we have a problem.
182  kodi::Log(ADDON_LOG_FATAL, "%s - fatal error creating thread - old thread id not null",
183  __func__);
184  exit(1);
185  }
186  }
187 
188  m_autoDelete = autoDelete;
189  m_threadStop = false;
190  m_startEvent.notify_all();
191  m_stopEvent.notify_all();
192 
193  std::promise<bool> prom;
194  m_future = prom.get_future();
195 
196  {
197  // The std::thread internals must be set prior to the lambda doing
198  // any work. This will cause the lambda to wait until m_thread
199  // is fully initialized. Interestingly, using a std::atomic doesn't
200  // have the appropriate memory barrier behavior to accomplish the
201  // same thing so a full system mutex needs to be used.
202  std::unique_lock<std::recursive_mutex> lock(m_threadMutex);
203  m_thread = new std::thread(
204  [](CThread* thread, std::promise<bool> promise) {
205  try
206  {
207  {
208  // Wait for the pThread->m_thread internals to be set. Otherwise we could
209  // get to a place where we're reading, say, the thread id inside this
210  // lambda's call stack prior to the thread that kicked off this lambda
211  // having it set. Once this lock is released, the CThread::Create function
212  // that kicked this off is done so everything should be set.
213  std::unique_lock<std::recursive_mutex> lock(thread->m_threadMutex);
214  }
215 
216  thread->m_threadId = std::this_thread::get_id();
217  std::stringstream ss;
218  ss << thread->m_threadId;
219  std::string id = ss.str();
220  bool autodelete = thread->m_autoDelete;
221 
222  kodi::Log(ADDON_LOG_DEBUG, "Thread %s start, auto delete: %s", id.c_str(),
223  (autodelete ? "true" : "false"));
224 
225  thread->m_running = true;
226  thread->m_startEvent.notify_one();
227 
228  thread->Process();
229 
230  if (autodelete)
231  {
232  kodi::Log(ADDON_LOG_DEBUG, "Thread %s terminating (autodelete)", id.c_str());
233  delete thread;
234  thread = nullptr;
235  }
236  else
237  kodi::Log(ADDON_LOG_DEBUG, "Thread %s terminating", id.c_str());
238  }
239  catch (const std::exception& e)
240  {
241  kodi::Log(ADDON_LOG_DEBUG, "Thread Terminating with Exception: %s", e.what());
242  }
243  catch (...)
244  {
245  kodi::Log(ADDON_LOG_DEBUG, "Thread Terminating with Exception");
246  }
247 
248  promise.set_value(true);
249  },
250  this, std::move(prom));
251 
252  m_startEvent.wait(lock);
253  }
254  }

265  void StopThread(bool wait = true)
266  {
267  std::unique_lock<std::recursive_mutex> lock(m_threadMutex);
268 
269  if (m_threadStop)
270  return;
271 
272  if (!m_running)
273  m_startEvent.wait(lock);
274  m_running = false;
275  m_threadStop = true;
276  m_stopEvent.notify_one();
277 
278  std::thread* lthread = m_thread;
279  if (lthread != nullptr && wait && !IsCurrentThread())
280  {
281  lock.unlock();
282  if (lthread->joinable())
283  lthread->join();
284  delete m_thread;
285  m_thread = nullptr;
286  m_threadId = std::thread::id();
287  }
288  }

328  bool Join(unsigned int milliseconds)
329  {
330  std::unique_lock<std::recursive_mutex> lock(m_threadMutex);
331  std::thread* lthread = m_thread;
332  if (lthread != nullptr)
333  {
334  if (IsCurrentThread())
335  return false;
336 
337  {
338  m_threadMutex.unlock(); // don't hold the thread lock while we're waiting
339  std::future_status stat = m_future.wait_for(std::chrono::milliseconds(milliseconds));
340  if (stat != std::future_status::ready)
341  return false;
342  m_threadMutex.lock();
343  }
344 
345  // it's possible it's already joined since we released the lock above.
346  if (lthread->joinable())
347  m_thread->join();
348  return true;
349  }
350  else
351  return false;
352  }
