937  inline std::string GetDirectoryName(const std::string& path)
938  {
939  // Will from a full filename return the directory the file resides in.
940  // Keeps the final slash at end and possible |option=foo options.
941 
942  size_t iPosSlash = path.find_last_of("/\\");
943  if (iPosSlash == std::string::npos)
944  return ""; // No slash, so no path (ignore any options)
945 
946  size_t iPosBar = path.rfind('|');
947  if (iPosBar == std::string::npos)
948  return path.substr(0, iPosSlash + 1); // Only path
949 
950  return path.substr(0, iPosSlash + 1) + path.substr(iPosBar); // Path + options
951  }

1074  inline bool StatFile(const std::string& filename, STAT_STRUCTURE& buffer)
1075  {
1076  struct __stat64 frontendBuffer = { };
1077  if (::kodi::addon::CAddonBase::m_interface->toKodi->kodi_filesystem->stat_file(::kodi::addon::CAddonBase::m_interface->toKodi->kodiBase, filename.c_str(), &frontendBuffer))
1078  {
1079  buffer.deviceId = frontendBuffer.st_dev;
1080  buffer.size = frontendBuffer.st_size;
1081 #if defined(TARGET_DARWIN) || defined(TARGET_FREEBSD)
1082  buffer.accessTime = frontendBuffer.st_atimespec;
1083  buffer.modificationTime = frontendBuffer.st_mtimespec;
1084  buffer.statusTime = frontendBuffer.st_ctimespec;
1085 #elif defined(TARGET_WINDOWS)
1086  buffer.accessTime = frontendBuffer.st_atime;
1087  buffer.modificationTime = frontendBuffer.st_mtime;
1088  buffer.statusTime = frontendBuffer.st_ctime;
1089 #elif defined(TARGET_ANDROID)
1090  buffer.accessTime.tv_sec = frontendBuffer.st_atime;
1091  buffer.accessTime.tv_nsec = frontendBuffer.st_atime_nsec;
1092  buffer.modificationTime.tv_sec = frontendBuffer.st_mtime;
1093  buffer.modificationTime.tv_nsec = frontendBuffer.st_mtime_nsec;
1094  buffer.statusTime.tv_sec = frontendBuffer.st_ctime;
1095  buffer.statusTime.tv_nsec = frontendBuffer.st_ctime_nsec;
1096 #else
1097  buffer.accessTime = frontendBuffer.st_atim;
1098  buffer.modificationTime = frontendBuffer.st_mtim;
1099  buffer.statusTime = frontendBuffer.st_ctim;
1100 #endif
1101  buffer.isDirectory = S_ISDIR(frontendBuffer.st_mode);
1102  buffer.isSymLink = S_ISLNK(frontendBuffer.st_mode);
1103  return true;
1104  }
1105  return false;
1106  }
