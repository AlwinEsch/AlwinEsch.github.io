892  inline std::string GetDirectoryName(const std::string& path)
893  {
894  // Will from a full filename return the directory the file resides in.
895  // Keeps the final slash at end and possible |option=foo options.
896 
897  size_t iPosSlash = path.find_last_of("/\\");
898  if (iPosSlash == std::string::npos)
899  return ""; // No slash, so no path (ignore any options)
900 
901  size_t iPosBar = path.rfind('|');
902  if (iPosBar == std::string::npos)
903  return path.substr(0, iPosSlash + 1); // Only path
904 
905  return path.substr(0, iPosSlash + 1) + path.substr(iPosBar); // Path + options
906  }

998  inline bool StatFile(const std::string& filename, STAT_STRUCTURE& buffer)
999  {
1000  struct __stat64 frontendBuffer = { };
1001  if (::kodi::addon::CAddonBase::m_interface->toKodi->kodi_filesystem->stat_file(::kodi::addon::CAddonBase::m_interface->toKodi->kodiBase, filename.c_str(), &frontendBuffer))
1002  {
1003  buffer.deviceId = frontendBuffer.st_dev;
1004  buffer.size = frontendBuffer.st_size;
1005 #if defined(TARGET_DARWIN) || defined(TARGET_FREEBSD)
1006  buffer.accessTime = frontendBuffer.st_atimespec;
1007  buffer.modificationTime = frontendBuffer.st_mtimespec;
1008  buffer.statusTime = frontendBuffer.st_ctimespec;
1009 #elif defined(TARGET_WINDOWS)
1010  buffer.accessTime = frontendBuffer.st_atime;
1011  buffer.modificationTime = frontendBuffer.st_mtime;
1012  buffer.statusTime = frontendBuffer.st_ctime;
1013 #elif defined(TARGET_ANDROID)
1014  buffer.accessTime.tv_sec = frontendBuffer.st_atime;
1015  buffer.accessTime.tv_nsec = frontendBuffer.st_atime_nsec;
1016  buffer.modificationTime.tv_sec = frontendBuffer.st_mtime;
1017  buffer.modificationTime.tv_nsec = frontendBuffer.st_mtime_nsec;
1018  buffer.statusTime.tv_sec = frontendBuffer.st_ctime;
1019  buffer.statusTime.tv_nsec = frontendBuffer.st_ctime_nsec;
1020 #else
1021  buffer.accessTime = frontendBuffer.st_atim;
1022  buffer.modificationTime = frontendBuffer.st_mtim;
1023  buffer.statusTime = frontendBuffer.st_ctim;
1024 #endif
1025  buffer.isDirectory = S_ISDIR(frontendBuffer.st_mode);
1026  buffer.isSymLink = S_ISLNK(frontendBuffer.st_mode);
1027  return true;
1028  }
1029  return false;
1030  }
