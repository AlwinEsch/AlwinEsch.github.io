939  inline std::string GetDirectoryName(const std::string& path)
940  {
941  // Will from a full filename return the directory the file resides in.
942  // Keeps the final slash at end and possible |option=foo options.
943 
944  size_t iPosSlash = path.find_last_of("/\\");
945  if (iPosSlash == std::string::npos)
946  return ""; // No slash, so no path (ignore any options)
947 
948  size_t iPosBar = path.rfind('|');
949  if (iPosBar == std::string::npos)
950  return path.substr(0, iPosSlash + 1); // Only path
951 
952  return path.substr(0, iPosSlash + 1) + path.substr(iPosBar); // Path + options
953  }

1076  inline bool StatFile(const std::string& filename, STAT_STRUCTURE& buffer)
1077  {
1078  struct __stat64 frontendBuffer = { };
1079  if (::kodi::addon::CAddonBase::m_interface->toKodi->kodi_filesystem->stat_file(::kodi::addon::CAddonBase::m_interface->toKodi->kodiBase, filename.c_str(), &frontendBuffer))
1080  {
1081  buffer.deviceId = frontendBuffer.st_dev;
1082  buffer.size = frontendBuffer.st_size;
1083 #if defined(TARGET_DARWIN) || defined(TARGET_FREEBSD)
1084  buffer.accessTime = frontendBuffer.st_atimespec;
1085  buffer.modificationTime = frontendBuffer.st_mtimespec;
1086  buffer.statusTime = frontendBuffer.st_ctimespec;
1087 #elif defined(TARGET_WINDOWS)
1088  buffer.accessTime = frontendBuffer.st_atime;
1089  buffer.modificationTime = frontendBuffer.st_mtime;
1090  buffer.statusTime = frontendBuffer.st_ctime;
1091 #elif defined(TARGET_ANDROID)
1092  buffer.accessTime.tv_sec = frontendBuffer.st_atime;
1093  buffer.accessTime.tv_nsec = frontendBuffer.st_atime_nsec;
1094  buffer.modificationTime.tv_sec = frontendBuffer.st_mtime;
1095  buffer.modificationTime.tv_nsec = frontendBuffer.st_mtime_nsec;
1096  buffer.statusTime.tv_sec = frontendBuffer.st_ctime;
1097  buffer.statusTime.tv_nsec = frontendBuffer.st_ctime_nsec;
1098 #else
1099  buffer.accessTime = frontendBuffer.st_atim;
1100  buffer.modificationTime = frontendBuffer.st_mtim;
1101  buffer.statusTime = frontendBuffer.st_ctim;
1102 #endif
1103  buffer.isDirectory = S_ISDIR(frontendBuffer.st_mode);
1104  buffer.isSymLink = S_ISLNK(frontendBuffer.st_mode);
1105  return true;
1106  }
1107  return false;
1108  }
