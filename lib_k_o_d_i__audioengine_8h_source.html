322  }
323  }
324 
330  bool RegisterMe(void* handle)
331  {
332  m_Handle = static_cast<AddonCB*>(handle);
333  if (m_Handle)
334  m_Callbacks = (KodiAPI::AudioEngine::CB_AudioEngineLib*)m_Handle->AudioEngineLib_RegisterMe(m_Handle->addonData);
335  if (!m_Callbacks)
336  fprintf(stderr, "libKODI_audioengine-ERROR: AudioEngineLib_RegisterMe can't get callback table from Kodi !!!\n");
337 
338  return m_Callbacks != nullptr;
339  }
340 
349  CAddonAEStream* MakeStream(AudioEngineFormat Format, unsigned int Options = 0)
350  {
351  AEStreamHandle *streamHandle = m_Callbacks->MakeStream(Format, Options);
352  if (!streamHandle)
353  {
354  fprintf(stderr, "libKODI_audioengine-ERROR: AudioEngine_make_stream MakeStream failed!\n");
355  return nullptr;
356  }
357 
358  return new CAddonAEStream(m_Handle, m_Callbacks, streamHandle);
359  }
