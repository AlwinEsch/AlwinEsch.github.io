1143  if (ret)

1146  static_cast<VFSDirEntry*>(malloc(sizeof(VFSDirEntry) * addonEntries.size()));
1147  for (unsigned int i = 0; i < addonEntries.size(); ++i)
1148  {
1149  entries[i].label = strdup(addonEntries[i].Label().c_str());
1150  entries[i].title = strdup(addonEntries[i].Title().c_str());
1151  entries[i].path = strdup(addonEntries[i].Path().c_str());
1152  entries[i].folder = addonEntries[i].IsFolder();
1153  entries[i].size = addonEntries[i].Size();
1154  entries[i].date_time = addonEntries[i].DateTime();
1155 
1156  entries[i].num_props = 0;
1157  const std::map<std::string, std::string>& props = addonEntries[i].GetProperties();
1158  if (!props.empty())
1159  {
1160  entries[i].properties =
1161  static_cast<VFSProperty*>(malloc(sizeof(VFSProperty) * props.size()));
1162  for (const auto& prop : props)
1163  {
1164  entries[i].properties[entries[i].num_props].name = strdup(prop.first.c_str());
1165  entries[i].properties[entries[i].num_props].val = strdup(prop.second.c_str());
1166  ++entries[i].num_props;
1167  }
1168  }
1169  else
