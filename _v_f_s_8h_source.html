1066  inline static bool ADDON_GetDirectory(const AddonInstance_VFSEntry* instance,
1067  const VFSURL* url,
1068  VFSDirEntry** retEntries,
1069  int* num_entries,
1070  VFSGetDirectoryCallbacks* callbacks)
1071  {
1072  std::vector<kodi::vfs::CDirEntry> addonEntries;
1073  bool ret = static_cast<CInstanceVFS*>(instance->toAddon->addonInstance)
1074  ->GetDirectory(*url, addonEntries, CVFSCallbacks(callbacks));
1075  if (ret)
1076  {
1077  VFSDirEntry* entries =
1078  static_cast<VFSDirEntry*>(malloc(sizeof(VFSDirEntry) * addonEntries.size()));
1079  for (unsigned int i = 0; i < addonEntries.size(); ++i)
1080  {
1081  entries[i].label = strdup(addonEntries[i].Label().c_str());
1082  entries[i].title = strdup(addonEntries[i].Title().c_str());
1083  entries[i].path = strdup(addonEntries[i].Path().c_str());
1084  entries[i].folder = addonEntries[i].IsFolder();
1085  entries[i].size = addonEntries[i].Size();
1086  entries[i].date_time = addonEntries[i].DateTime();
1087 
1088  entries[i].num_props = 0;
1089  const std::map<std::string, std::string>& props = addonEntries[i].GetProperties();
1090  if (!props.empty())
1091  {
1092  entries[i].properties =
1093  static_cast<VFSProperty*>(malloc(sizeof(VFSProperty) * props.size()));
1094  for (const auto& prop : props)
1095  {
1096  entries[i].properties[entries[i].num_props].name = strdup(prop.first.c_str());
1097  entries[i].properties[entries[i].num_props].val = strdup(prop.second.c_str());
1098  ++entries[i].num_props;
1099  }
1100  }
1101  else
1102  entries[i].properties = nullptr;
1103  }
1104  *retEntries = entries;
1105  *num_entries = static_cast<int>(addonEntries.size());
1106  }
