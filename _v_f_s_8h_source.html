1226  inline std::string GetDirectoryName(const std::string& path)
1227  {
1228  // Will from a full filename return the directory the file resides in.
1229  // Keeps the final slash at end and possible |option=foo options.
1230 
1231  size_t iPosSlash = path.find_last_of("/\\");
1232  if (iPosSlash == std::string::npos)
1233  return ""; // No slash, so no path (ignore any options)
1234 
1235  size_t iPosBar = path.rfind('|');
1236  if (iPosBar == std::string::npos)
1237  return path.substr(0, iPosSlash + 1); // Only path
1238 
1239  return path.substr(0, iPosSlash + 1) + path.substr(iPosBar); // Path + options
1240  }
