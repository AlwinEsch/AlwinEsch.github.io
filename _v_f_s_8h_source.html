505  inline static bool ADDON_GetDirectory(const AddonInstance_VFSEntry* instance,
506  const VFSURL* url,
507  VFSDirEntry** retEntries,
508  int* num_entries,
509  VFSGetDirectoryCallbacks* callbacks)
510  {
511  std::vector<kodi::vfs::CDirEntry> addonEntries;
512  bool ret = instance->toAddon.addonInstance->GetDirectory(*url, addonEntries, CVFSCallbacks(callbacks));
513  if (ret)
514  {
515  VFSDirEntry* entries = static_cast<VFSDirEntry*>(malloc(sizeof(VFSDirEntry) * addonEntries.size()));
516  for (unsigned int i = 0; i < addonEntries.size(); ++i)
517  {
518  entries[i].label = strdup(addonEntries[i].Label().c_str());
519  entries[i].title = strdup(addonEntries[i].Title().c_str());
520  entries[i].path = strdup(addonEntries[i].Path().c_str());
521  entries[i].folder = addonEntries[i].IsFolder();
522  entries[i].size = addonEntries[i].Size();
523  entries[i].date_time = addonEntries[i].DateTime();
524 
525  entries[i].num_props = 0;
526  const std::map<std::string, std::string>& props = addonEntries[i].GetProperties();
527  if (!props.empty())
528  {
529  entries[i].properties = static_cast<VFSProperty*>(malloc(sizeof(VFSProperty)*props.size()));
530  for (const auto& prop : props)
531  {
532  entries[i].properties[entries[i].num_props].name = strdup(prop.first.c_str());
533  entries[i].properties[entries[i].num_props].val = strdup(prop.second.c_str());
534  ++entries[i].num_props;
535  }
536  }
537  else
538  entries[i].properties = nullptr;
539  }
540  *retEntries = entries;
541  *num_entries = static_cast<int>(addonEntries.size());
542  }
543  return ret;
544  }
