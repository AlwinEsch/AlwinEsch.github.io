1243  inline std::string GetDirectoryName(const std::string& path)
1244  {
1245  // Will from a full filename return the directory the file resides in.
1246  // Keeps the final slash at end and possible |option=foo options.
1247 
1248  size_t iPosSlash = path.find_last_of("/\\");
1249  if (iPosSlash == std::string::npos)
1250  return ""; // No slash, so no path (ignore any options)
1251 
1252  size_t iPosBar = path.rfind('|');
1253  if (iPosBar == std::string::npos)
1254  return path.substr(0, iPosSlash + 1); // Only path
1255 
1256  return path.substr(0, iPosSlash + 1) + path.substr(iPosBar); // Path + options
1257  }
