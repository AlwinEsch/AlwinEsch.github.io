385 {

390  bool bottom_left_origin; // Use conventional bottom-left origin convention. Is false, standard top-left origin semantics are used
391  unsigned version_major; // Major version number for core GL context
392  unsigned version_minor; // Minor version number for core GL context
393  bool cache_context; // If this is true, the frontend will go very far to avoid resetting context in scenarios like toggling fullscreen, etc.
394  // The reset callback might still be called in extreme situations such as if the context is lost beyond recovery
395  // For optimal stability, set this to false, and allow context to be reset at any time
