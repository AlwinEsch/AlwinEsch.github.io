325 class AudioEngineFormat : public addon::CStructHdl<AudioEngineFormat, AUDIO_ENGINE_FORMAT>
326 {
327 public:
329  AudioEngineFormat()
330  {
331  m_cStructure->m_dataFormat = AUDIOENGINE_FMT_INVALID;
332  m_cStructure->m_sampleRate = 0;
333  m_cStructure->m_encodedRate = 0;
334  m_cStructure->m_frames = 0;
335  m_cStructure->m_frameSize = 0;
336  m_cStructure->m_channelCount = 0;
337 
338  for (size_t ch = 0; ch < AUDIOENGINE_CH_MAX; ++ch)
339  m_cStructure->m_channels[ch] = AUDIOENGINE_CH_NULL;
340  }
341  AudioEngineFormat(const AudioEngineFormat& channel) : CStructHdl(channel) {}
342  AudioEngineFormat(const AUDIO_ENGINE_FORMAT* channel) : CStructHdl(channel) {}
343  AudioEngineFormat(AUDIO_ENGINE_FORMAT* channel) : CStructHdl(channel) {}
346 
365 
366 
368  void SetDataFormat(enum AudioEngineDataFormat format) { m_cStructure->m_dataFormat = format; }
369 
371  enum AudioEngineDataFormat GetDataFormat() const { return m_cStructure->m_dataFormat; }
372 
374  void SetSampleRate(unsigned int rate) { m_cStructure->m_sampleRate = rate; }
375 
377  unsigned int GetSampleRate() const { return m_cStructure->m_sampleRate; }
378 
380  void SetEncodedRate(unsigned int rate) { m_cStructure->m_encodedRate = rate; }
381 
383  unsigned int GetEncodedRate() const { return m_cStructure->m_encodedRate; }
384 
386  void SetChannelLayout(const std::vector<enum AudioEngineChannel>& layout)
387  {
388  // Reset first all to empty values to AUDIOENGINE_CH_NULL, in case given list is empty
389  m_cStructure->m_channelCount = 0;
390  for (size_t ch = 0; ch < AUDIOENGINE_CH_MAX; ++ch)
391  m_cStructure->m_channels[ch] = AUDIOENGINE_CH_NULL;
392 
393  for (size_t ch = 0; ch < layout.size() && ch < AUDIOENGINE_CH_MAX; ++ch)
394  {
395  m_cStructure->m_channels[ch] = layout[ch];
396  m_cStructure->m_channelCount++;
397  }
398  }
399 
401  std::vector<enum AudioEngineChannel> GetChannelLayout() const
402  {
403  std::vector<enum AudioEngineChannel> channels;
404  for (size_t ch = 0; ch < AUDIOENGINE_CH_MAX; ++ch)
405  {
406  if (m_cStructure->m_channels[ch] == AUDIOENGINE_CH_NULL)
407  break;
408 
409  channels.push_back(m_cStructure->m_channels[ch]);
410  }
411  return channels;
412  }
413 
415  void SetFramesAmount(unsigned int frames) { m_cStructure->m_frames = frames; }
416 
418  unsigned int GetFramesAmount() const { return m_cStructure->m_frames; }
419 
421  void SetFrameSize(unsigned int frameSize) { m_cStructure->m_frameSize = frameSize; }
422 
424  unsigned int GetFrameSize() const { return m_cStructure->m_frameSize; }
425 
427  bool CompareFormat(const AudioEngineFormat* fmt)
428  {
429  if (!fmt)
430  {
431  return false;
432  }
433 
434  if (m_cStructure->m_dataFormat != fmt->m_cStructure->m_dataFormat ||
435  m_cStructure->m_sampleRate != fmt->m_cStructure->m_sampleRate ||
436  m_cStructure->m_encodedRate != fmt->m_cStructure->m_encodedRate ||
437  m_cStructure->m_frames != fmt->m_cStructure->m_frames ||
438  m_cStructure->m_frameSize != fmt->m_cStructure->m_frameSize ||
439  m_cStructure->m_channelCount != fmt->m_cStructure->m_channelCount)
440  {
441  return false;
442  }
443 
444  for (unsigned int ch = 0; ch < AUDIOENGINE_CH_MAX; ++ch)
445  {
446  if (fmt->m_cStructure->m_channels[ch] != m_cStructure->m_channels[ch])
447  {
448  return false;
449  }
450  }
451 
452  return true;
453  }
454 };
