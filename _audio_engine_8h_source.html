64 struct AudioEngineFormat
65 {

68 
70  unsigned int m_sampleRate;

80

83 
85  unsigned int m_frameSize;
86 
87  AudioEngineFormat()
88  {
89  m_dataFormat = AE_FMT_INVALID;
90  m_sampleRate = 0;
91  m_encodedRate = 0;
92  m_frames = 0;
93  m_frameSize = 0;
94  m_channelCount = 0;
95 
96  for (unsigned int ch = 0; ch < AE_CH_MAX; ++ch)
97  {
98  m_channels[ch] = AE_CH_MAX;
99  }
100  }
101 
103  bool compareFormat(const AudioEngineFormat *fmt)
104  {
105  if (!fmt)
106  {
107  return false;
108  }
109 
110  if (m_dataFormat != fmt->m_dataFormat ||
111  m_sampleRate != fmt->m_sampleRate ||
112  m_encodedRate != fmt->m_encodedRate ||
113  m_frames != fmt->m_frames ||
114  m_frameSize != fmt->m_frameSize ||
115  m_channelCount != fmt->m_channelCount)
116  {
117  return false;
118  }
119 
120  for (unsigned int ch = 0; ch < AE_CH_MAX; ++ch)
121  {
122  if (fmt->m_channels[ch] != m_channels[ch])
123  {
124  return false;
125  }
126  }
127 
128  return true;
129  }
130 };
