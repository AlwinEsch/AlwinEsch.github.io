145  bool compareFormat(const AudioEngineFormat *fmt)
146  {
147  if (!fmt)
148  {
149  return false;
150  }
151 
152  if (m_dataFormat != fmt->m_dataFormat ||
153  m_sampleRate != fmt->m_sampleRate ||
154  m_encodedRate != fmt->m_encodedRate ||
155  m_frames != fmt->m_frames ||
156  m_frameSize != fmt->m_frameSize ||
157  m_channelCount != fmt->m_channelCount)
158  {
159  return false;
160  }
161 
162  for (unsigned int ch = 0; ch < AUDIO_CH_MAX; ch++)
163  {
164  if (fmt->m_channels[ch] != m_channels[ch])
165  {
166  return false;
167  }
168  }
169 
170  return true;
171  }
172  } AudioEngineFormat;
