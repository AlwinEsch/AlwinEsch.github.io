128  AudioEngineFormat()
129  {

131  m_sampleRate = 0;

133  m_frames = 0;
134  m_frameSize = 0;
135  m_channelCount = 0;

144  bool compareFormat(const AudioEngineFormat *fmt)
145  {
146  if (!fmt)
147  {
148  return false;
149  }
150 
151  if (m_dataFormat != fmt->m_dataFormat ||
152  m_sampleRate != fmt->m_sampleRate ||
153  m_encodedRate != fmt->m_encodedRate ||
154  m_frames != fmt->m_frames ||
155  m_frameSize != fmt->m_frameSize ||
156  m_channelCount != fmt->m_channelCount)
157  {
158  return false;
159  }
160 
161  for (unsigned int ch = 0; ch < AUDIO_CH_MAX; ch++)
162  {
163  if (fmt->m_channels[ch] != m_channels[ch])
164  {
165  return false;
166  }
167  }
168 
169  return true;
170  }
171  };
