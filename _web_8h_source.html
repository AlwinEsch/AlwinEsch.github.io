545  inline static bool control_get_history(const AddonInstance_WebAddon* instance, const ADDON_HANDLE handle,
546  char*** list, unsigned int* entries, bool behind_current)
547  {
548  std::vector<std::string> historyWebsiteNames;
549  bool ret = static_cast<CWebControl*>(handle->callerAddress)->GetHistory(historyWebsiteNames, behind_current);
550  if (ret)
551  {
552  *list = static_cast<char**>(malloc(historyWebsiteNames.size()*sizeof(char*)));
553  *entries = static_cast<unsigned int>(historyWebsiteNames.size());
554  for (size_t i = 0; i < historyWebsiteNames.size(); ++i)
555 #ifdef WIN32 // To prevent warning C4996
556  (*list)[i] = _strdup(historyWebsiteNames[i].c_str());
557 #else
558  (*list)[i] = strdup(historyWebsiteNames[i].c_str());
559 #endif
560  }
561  return ret;
562  }
