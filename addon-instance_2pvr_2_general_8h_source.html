364  m_capabilities->recordingsLifetimeValues[i].iValue =
365  recordingsLifetimeValues[i].GetCStructure()->iValue;
366  strncpy(m_capabilities->recordingsLifetimeValues[i].strDescription,
367  recordingsLifetimeValues[i].GetCStructure()->strDescription,
368  sizeof(m_capabilities->recordingsLifetimeValues[i].strDescription) - 1);
369  ++m_capabilities->iRecordingsLifetimesSize;
370  }
371  }

374  std::vector<PVRRecordingLifetimeValue> GetRecordingsLifetimeValues() const
375  {

377  for (unsigned int i = 0; i < m_capabilities->iRecordingsLifetimesSize; ++i)
378  recordingsLifetimeValues.emplace_back(
379  m_capabilities->recordingsLifetimeValues[i].iValue,
