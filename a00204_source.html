199  static CodecDescriptor GetCodecByName(const char* strCodecName)
200  {
201  CodecDescriptor retVal;
202  // some of the backends codec names don't match ffmpeg's, so translate them to something ffmpeg understands
203  if (!strcmp(strCodecName, "MPEG2AUDIO"))
204  retVal = CodecDescriptor(KodiAPI::InputStream::GetCodecByName("MP2"), strCodecName);
205  else if (!strcmp(strCodecName, "MPEGTS"))
206  retVal = CodecDescriptor(KodiAPI::InputStream::GetCodecByName("MPEG2VIDEO"), strCodecName);
207  else if (!strcmp(strCodecName, "TEXTSUB"))
208  retVal = CodecDescriptor(KodiAPI::InputStream::GetCodecByName("TEXT"), strCodecName);
209  else
210  retVal = CodecDescriptor(KodiAPI::InputStream::GetCodecByName(strCodecName), strCodecName);
211 
212  return retVal;
213  }
