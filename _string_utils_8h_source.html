586  {
587  if (!fmt || !fmt[0])
588  return "";
589 
590  int size = FORMAT_BLOCK_SIZE;
591  va_list argCopy;
592 
593  while (true)
594  {
595  char* cstr = reinterpret_cast<char*>(malloc(sizeof(char) * size));
596  if (!cstr)
597  return "";
598 
599  va_copy(argCopy, args);
600  int nActual = vsnprintf(cstr, size, fmt, argCopy);
601  va_end(argCopy);
602 
603  if (nActual > -1 && nActual < size) // We got a valid result
604  {
605  std::string str(cstr, nActual);
606  free(cstr);
607  return str;
608  }
609  free(cstr);
610 #ifndef TARGET_WINDOWS
611  if (nActual > -1) // Exactly what we will need (glibc 2.1)

613  else // Let's try to double the size (glibc 2.0)
614  size *= 2;
615 #else // TARGET_WINDOWS
