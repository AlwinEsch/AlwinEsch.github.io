340  template <class THE_CLASS, typename THE_STRUCT>
341  class PeripheralVector
342  {
343  public:
344  static void ToStructs(const std::vector<THE_CLASS>& vecObjects, THE_STRUCT** pStructs)
345  {
346  if (!pStructs)
347  return;
348 
349  if (vecObjects.empty())
350  {
351  *pStructs = NULL;
352  }
353  else
354  {
355  (*pStructs) = new THE_STRUCT[vecObjects.size()];
356  for (unsigned int i = 0; i < vecObjects.size(); i++)
357  vecObjects.at(i).ToStruct((*pStructs)[i]);
358  }
359  }
360 
361  static void ToStructs(const std::vector<THE_CLASS*>& vecObjects, THE_STRUCT** pStructs)
362  {
363  if (!pStructs)
364  return;
365 
366  if (vecObjects.empty())
367  {
368  *pStructs = NULL;
369  }
370  else
371  {
372  *pStructs = new THE_STRUCT[vecObjects.size()];
373  for (unsigned int i = 0; i < vecObjects.size(); i++)
374  vecObjects.at(i)->ToStruct((*pStructs)[i]);
375  }
376  }
377 
378  static void FreeStructs(unsigned int structCount, THE_STRUCT* structs)
379  {
380  if (structs)
381  {
382  for (unsigned int i = 0; i < structCount; i++)
383  THE_CLASS::FreeStruct(structs[i]);
384  }
385  PERIPHERAL_SAFE_DELETE_ARRAY(structs);
386  }
387  };
