47  inline bool CVFSFile::OpenFileForWrite(const std::string& strFileName, bool bOverWrite)
48  {
49  Close();
50 
51  // Try to open the file. If it fails, check if we need to create the directory first
52  // This way we avoid checking if the directory exists every time
53  m_pFile = CAddonLib::g_libKODI_Internal->OpenFileForWrite(strFileName.c_str(), bOverWrite);
54  if (!m_pFile)
55  {
56  std::string cacheDirectory = GetDirectoryName(strFileName);
57  if (CAddonLib::g_libKODI_Internal->DirectoryExists(cacheDirectory.c_str()) || m_XBMC->CreateDirectory(cacheDirectory.c_str()))
58  m_pFile = CAddonLib::g_libKODI_Internal->OpenFileForWrite(strFileName.c_str(), bOverWrite);
59  }
60  return m_pFile != nullptr;
61  }

70  inline bool CVFSFile::ReadLine(std::string &strLine)
71  {
72  strLine.clear();
73  if (!m_pFile)
74  return false;
75  // TODO: Read 1024 chars into buffer. If file position advanced that many
76  // chars, we didn't hit a newline. Otherwise, if file position is 1 or 2
77  // past the number of bytes read, we read (and skipped) a newline sequence.
78  char buffer[1025];
79  if (CAddonLib::g_libKODI_Internal->ReadFileString(m_pFile, buffer, sizeof(buffer))) // will read 1024 bytes
80  {
81  strLine = buffer;
82  return !strLine.empty();
83  }
84  return false;
85  }
