192  bool RegisterMe(void *handle)
193  {
194  m_Handle = static_cast<AddonCB*>(handle);
195  if (m_Handle)
196  m_Callbacks = (KodiAPI::AddOn::CB_AddOnLib*)m_Handle->AddOnLib_RegisterMe(m_Handle->addonData);
197  if (!m_Callbacks)
198  fprintf(stderr, "libXBMC_addon-ERROR: AddOnLib_RegisterMe can't get callback table from Kodi !!!\n");
199 
200  return m_Callbacks != nullptr;
201  }
