195  bool RegisterMe(void *handle)
196  {
197  m_Handle = static_cast<AddonCB*>(handle);
198  if (m_Handle)
199  m_Callbacks = (KodiAPI::AddOn::CB_AddOnLib*)m_Handle->AddOnLib_RegisterMe(m_Handle->addonData);
200  if (!m_Callbacks)
201  fprintf(stderr, "libXBMC_addon-ERROR: AddOnLib_RegisterMe can't get callback table from Kodi !!!\n");
202 
203  return m_Callbacks != nullptr;
204  }
