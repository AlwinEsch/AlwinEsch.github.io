148  bool RegisterMe(void *handle)
149  {
150  m_Handle = static_cast<AddonCB*>(handle);
151  if (m_Handle)
152  m_Callbacks = (KodiAPI::AddOn::CB_AddOnLib*)m_Handle->AddOnLib_RegisterMe(m_Handle->addonData);
153  if (!m_Callbacks)
154  fprintf(stderr, "libXBMC_addon-ERROR: AddOnLib_RegisterMe can't get callback table from Kodi !!!\n");
155 
156  return m_Callbacks != nullptr;
157  }

166  void Log(const addon_log_t loglevel, const char *format, ... )
167  {
168  char buffer[16384];
169  static constexpr size_t len = sizeof (buffer) - 1;
170  va_list args;
171  va_start (args, format);
172  vsnprintf (buffer, len, format, args);
173  va_end (args);
174  buffer[len] = '\0'; // to be sure it's null-terminated
175  m_Callbacks->Log(m_Handle->addonData, loglevel, buffer);
176  }
