175  bool RegisterMe(void *handle)
176  {
177  m_Handle = static_cast<AddonCB*>(handle);
178  if (m_Handle)
179  m_Callbacks = (KodiAPI::AddOn::CB_AddOnLib*)m_Handle->AddOnLib_RegisterMe(m_Handle->addonData);
180  if (!m_Callbacks)
181  fprintf(stderr, "libXBMC_addon-ERROR: AddOnLib_RegisterMe can't get callback table from Kodi !!!\n");
182 
183  return m_Callbacks != nullptr;
184  }

193  void Log(const addon_log_t loglevel, const char *format, ... )
194  {
195  char buffer[16384];
196  static constexpr size_t len = sizeof (buffer) - 1;
197  va_list args;
198  va_start (args, format);
199  vsnprintf (buffer, len, format, args);
200  va_end (args);
201  buffer[len] = '\0'; // to be sure it's null-terminated
202  m_Callbacks->Log(m_Handle->addonData, loglevel, buffer);
203  }
