791 inline bool GetKeyboardLayout(int modifierKey, std::string& layout_name, std::vector<std::vector<std::string>>& layout)
792 {
793  AddonToKodiFuncTable_Addon* toKodi = ::kodi::addon::CAddonBase::m_interface->toKodi;
794  AddonKeyboardKeyTable c_layout;
795  char* c_layout_name = nullptr;
796  bool ret = toKodi->kodi->get_keyboard_layout(toKodi->kodiBase, &c_layout_name, modifierKey, &c_layout);
797  if (ret)
798  {
799  if (c_layout_name)
800  {
801  layout_name = c_layout_name;
802  toKodi->free_string(toKodi->kodiBase, c_layout_name);
803  }
804 
805  layout.resize(STD_KB_BUTTONS_MAX_ROWS);
806  for (unsigned int row = 0; row < STD_KB_BUTTONS_MAX_ROWS; row++)
807  {
808  layout[row].resize(STD_KB_BUTTONS_PER_ROW);
809  for (unsigned int column = 0; column < STD_KB_BUTTONS_PER_ROW; column++)
810  {
811  char* button = c_layout.keys[row][column];
812  if (button)
813  {
814  layout[row][column] = button;
815  toKodi->free_string(toKodi->kodiBase, button);
816  }
817  }
818  }
819  }
820  return ret;
821 }
822 } /* namespace kodi */
