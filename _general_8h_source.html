834 inline bool GetKeyboardLayout(int modifierKey, std::string& layout_name, std::vector<std::vector<std::string>>& layout)
835 {
836  AddonToKodiFuncTable_Addon* toKodi = ::kodi::addon::CAddonBase::m_interface->toKodi;
837  AddonKeyboardKeyTable c_layout;
838  char* c_layout_name = nullptr;
839  bool ret = toKodi->kodi->get_keyboard_layout(toKodi->kodiBase, &c_layout_name, modifierKey, &c_layout);
840  if (ret)
841  {
842  if (c_layout_name)
843  {
844  layout_name = c_layout_name;
845  toKodi->free_string(toKodi->kodiBase, c_layout_name);
846  }
847 
848  layout.resize(STD_KB_BUTTONS_MAX_ROWS);
849  for (unsigned int row = 0; row < STD_KB_BUTTONS_MAX_ROWS; row++)
850  {
851  layout[row].resize(STD_KB_BUTTONS_PER_ROW);
852  for (unsigned int column = 0; column < STD_KB_BUTTONS_PER_ROW; column++)
853  {
854  char* button = c_layout.keys[row][column];
855  if (button)
856  {
857  layout[row][column] = button;
858  toKodi->free_string(toKodi->kodiBase, button);
859  }
860  }
861  }
862  }
863  return ret;
864 }
